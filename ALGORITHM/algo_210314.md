def solution(a, b):
    answer = ''
    
    # 29: 2
    # 31: 1, 3, 5, 7, 8, 10, 12
    # 30: 4, 6, 9, 11
    month = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    day = ['FRI', 'SAT', 'SUN', 'MON', 'TUE', 'WED', 'THU']
    daySum = 0
    for i in range(a - 1):
        daySum += month[i]
        
    result = (daySum + b) % 7
    return day[result - 1]
